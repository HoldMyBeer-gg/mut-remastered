**Add world-command send/recv helpers to TestClient:**
```rust
pub async fn send_world(&mut self, msg: &protocol::world::ClientMsg) {
    let bytes = encode_message(msg).expect("encode world ClientMsg");
    self.stream.write_all(&bytes).await.expect("write world msg");
}

pub async fn recv_world(&mut self) -> protocol::world::ServerMsg {
    let mut len_buf = [0u8; 4];
    self.stream.read_exact(&mut len_buf).await.expect("read len");
    let payload_len = u32::from_le_bytes(len_buf) as usize;
    let mut payload = vec![0u8; payload_len];
    self.stream.read_exact(&mut payload).await.expect("read payload");
    decode_message::<protocol::world::ServerMsg>(&payload).expect("decode world ServerMsg")
}
```

**Add a helper method for login flow** (register + login, returns account_id):
```rust
pub async fn register_and_login(&mut self, username: &str, password: &str) -> String {
    self.send(&protocol::auth::ClientMsg::Register {
        username: username.to_string(),
        password: password.to_string(),
    }).await;
    let resp = self.recv().await;
    // expect RegisterOk
    self.send(&protocol::auth::ClientMsg::Login {
        username: username.to_string(),
        password: password.to_string(),
    }).await;
    let resp = self.recv().await;
    match resp {
        protocol::auth::ServerMsg::LoginOk { session_token } => session_token,
        other => panic!("expected LoginOk, got {:?}", other),
    }
}
```
